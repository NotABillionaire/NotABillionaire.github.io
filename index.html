import requests

URL = "https://api.dictionaryapi.dev/api/v2/entries/en/"

# Search function to handle the dictionary lookup
def search():
    word = Element("word").value  # Get the word entered by the user
    output = Element("output")  # Reference the output element
    
    if not word:
        output.write("Please enter a word to search.\n")
        return

    try:
        response = requests.get(URL + word)
        
        # Check for 404 status (word not found)
        if response.status_code == 404:
            output.write(f"{word} is not a valid word.\n")
            return

        content = response.json()
        contents = content[0]

        result = f"Word: {contents['word']}\n"
        phonetics = contents["phonetics"][0]["audio"]
        result += f"Audio: {phonetics}\n\n"

        meanings = contents["meanings"]
        for meaning in meanings:
            result += f"{meaning['partOfSpeech']}: \n"
            definitions = meaning["definitions"]
            for definition in definitions:
                result += f" - Definition: {definition['definition']}\n"
            synonyms = ", ".join(meaning["synonyms"])
            antonyms = ", ".join(meaning["antonyms"])
            result += f" - Synonyms: {synonyms}\n"
            result += f" - Antonyms: {antonyms}\n\n"

        output.write(result)
    except Exception as e:
        output.write(f"An error occurred: {e}")

# Attach search function to button click
Element("searchButton").element.onclick = search
